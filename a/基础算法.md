## 排序

### 快速排序

```cpp
#include <iostream>
using namespace std;

const int N = 1e6 + 10;
int n;
int q[N];

void qs(int *q, int l, int r) {
    if (l >= r) return ;
    int x = q[l], i = l - 1, j = r + 1;
    while (i < j) {
        while (q[++i] < x);
        while (q[--j] < x);
        if (i < j) swap(q[i], q[j]);
    }
    qs(q, l, j);
    qs(q, j + 1, r);
}

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &q[i]);
    qs(q, 0, n - 1);
    
    for (int i = 0; i < n; i++) printf("%d ", q[i]);
}
```

为啥循环条件不能等于呀? `do i++; while(a[i] <= x);`

> 会陷入死循环，比如2 1，2 <= 2、1 <= 2，0 <= 2，会不断的加。

为什么不能`qs(q, l, j - 1); qs(q, j, r);`?

> 当区间长度是2的时候，可能有边界问题。例如 2 1，那么会一直递归，因为区间一直没有压缩，也就是一直在[0,1]。可以写成qs(q, l, i - 1); qs(q,i, r);，但是相应的x=q[r]。

为什么不能`i=l, j=r`呢？

> 会出现边界问题，例如 99 51 84 48 76 42 6 6 38 10。会一直递归下去。

### 归并排序

```cpp
#include <iostream>
using namespace std;

const int N = 1e6 + 10;
int n;
int q[N], tmp[N];

void ms(int *q, int l, int r) {
    if (l >= r) return ;
    
    int mid = l + r >> 1;
    ms(q, l, mid), ms(q, mid + 1, r);
    int k = 0, i = l, j = mid + 1;
    
    while (i <= mid && j <= r) {
        if (q[i] <= q[j]) tmp[k++] = q[i++];
        else tmp[k++] = q[j++];
    }
    while (i <= mid) tmp[k++] = q[i++];
    while (j <= r) tmp[k++] = q[j++];
    
    for (i = l, j = 0; i <= r; i++, j++) q[i] = tmp[j];
}

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &q[i]);
    ms(q, 0, n - 1);
    
    for (int i = 0; i < n; i++) printf("%d ", q[i]);
}
```

## 高精度

### 加法

```cpp
#include <iostream>
#include <vector>

using namespace std;

vector<int> add(vector<int>& a, vector<int>& b) {
    vector<int> c;
    int t = 0;
    for (int i = 0; i < a.size() || i < b.size(); i++) {
        if (i < a.size()) t += a[i];
        if (i < b.size()) t += b[i];
        c.emplace_back(t % 10);
        t /= 10;
    }
    if (t) c.emplace_back(1);
    return c;
}

int main() {
    string a, b;
    vector<int> A, B;
    cin >> a >> b;
    for (int i = a.size() - 1; ~i; i--) A.emplace_back(a[i] - '0');
    for (int i = b.size() - 1; ~i; i--) B.emplace_back(b[i] - '0');
    
    auto C = add(A, B);
    for (int i = C.size() - 1; ~i; i--) printf("%d", C[i]);
}
```

### 减法

```cpp
#include <iostream>
#include <vector>
using namespace std;

bool cmp(vector<int>& a, vector<int>& b) {
    if (a.size() != b.size()) return a.size() > b.size();
    for (int i = a.size() - 1; ~i; i--)  {
        if (a[i] != b[i]) return a[i] > b[i];
    }
    return true;
}

vector<int> sub(vector<int>& a, vector<int>& b) {
    vector<int> c;
    for (int i = 0, t = 0; i < a.size(); i++) {
        t = a[i] - t;
        if (i < b.size()) t -= b[i];
        c.emplace_back((t + 10) % 10);
        if (t < 0) t = 1;
        else t = 0;
    }
    
    while (c.size() > 1 && c.back() == 0) c.pop_back();
    return c;
}

int main() {
    string a, b;
    cin >> a >> b;
    vector<int> A, B, C;
    for (int i = a.size() - 1; ~i; i--) A.emplace_back(a[i] - '0');
    for (int i = b.size() - 1; ~i; i--) B.emplace_back(b[i] - '0');
    
    if (cmp(A, B)) {
        C = sub(A, B);
    } else {
        C = sub(B, A);
        printf("-");
    }
    for (int i = C.size() - 1; ~i; i--) printf("%d", C[i]);
}
```

### 除法

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

vector<int> div(vector<int>& a, int b, int& r) {
    vector<int> c;
    r = 0;
    for (int i = a.size() - 1; ~i; i--) {
        r = r * 10 + a[i];
        c.emplace_back(r / b);
        r %= b;
    }
    
    reverse(c.begin(), c.end());
    while (c.size() > 1 && c.back() == 0) c.pop_back();
    return c;
}

int main() {
    string a;
    int b, r;
    cin >> a >> b;
    vector<int> A, C;
    
    for (int i = a.size() - 1; ~i; i--) A.emplace_back(a[i] - '0');
    C = div(A, b, r);
    for (int i = C.size() - 1; ~i; i--) printf("%d", C[i]);
    printf("\n%d", r);
}
```

## 前缀和差分

### 一维前缀和

```cpp
#include <iostream>
using namespace std;

const int N = 1e6 + 10;

int n, m;
int q[N], s[N];
int main() {
    scanf("%d %d", &n, &m);
    for (int i = 1; i <= n; i++) scanf("%d", &q[i]);
    for (int i = 1; i <= n; i++) s[i] = s[i - 1] + q[i];
    
    for (int i = 0; i < m; i++) {
        int l, r;
        scanf("%d %d", &l, &r);
        printf("%d\n", s[r] - s[l - 1]);
    }
}
```

### 二维前缀和

```cpp
#include <iostream>
using namespace std;

int n, m, q;
const int N = 1010;
int a[N][N], s[N][N];

int main() {
    scanf("%d%d%d", &n, &m, &q);
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            scanf("%d", &a[i][j]);
        }
    }
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            s[i][j] = s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1] + a[i][j];
        }
    }
    
    for (int i = 0; i < q; i++) {
        int x1, y1, x2, y2;
        scanf("%d%d%d%d", &x1, &y1, &x2, &y2);
        printf("%d\n", s[x2][y2] - s[x1 - 1][y2] - s[x2][y1 - 1] + s[x1 - 1][y1 - 1]);
    }
}
```

### 一维差分

```cpp
#include <iostream>
using namespace std;

const int N = 1e6 + 10;
int n, m;
int a[N], b[N];

void insert(int l, int r, int c) {
    b[l] += c;
    b[r + 1] -= c;
}

int main() {
    scanf("%d %d", &n, &m); 
    for (int i = 1; i <= n; i++) scanf("%d", &a[i]);
    for (int i = 1; i <= n; i++) insert(i, i, a[i]);
    
    int l, r, c;
    for (int i = 0; i < m; i++) {
        scanf("%d %d %d", &l, &r, &c);
        insert(l, r, c);
    }
    
    for (int i = 1; i <= n; i++) b[i] += b[i - 1];
    for (int i = 1; i <= n; i++) printf("%d ", b[i]);
}
```

### 二维差分

```cpp
#include <iostream>
using namespace std;

const int N = 1010;
int n, m, q;
int a[N][N], b[N][N];

void insert(int x1, int y1, int x2, int y2, int c) {
    b[x1][y1] += c;
    b[x2 + 1][y1] -= c;
    b[x1][y2 + 1] -= c;
    b[x2 + 1][y2 + 1] += c;
}

int main() {
    scanf("%d %d %d", &n, &m, &q);
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            scanf("%d", &a[i][j]);
            insert(i, j, i, j, a[i][j]);
        }
    }
    
    int x1, y1, x2, y2, c;
    for (int i = 0; i < q; i++) {
        scanf("%d %d %d %d %d", &x1, &y1, &x2, &y2, &c);
        insert(x1, y1, x2, y2, c);
    }
    
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            b[i][j] += b[i - 1][j] + b[i][j - 1] - b[i - 1][j - 1];
        }
    }
    
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            printf("%d ", b[i][j]);
        }
        puts("");
    }
}
```
