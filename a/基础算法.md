## 排序

### 快速排序

```cpp
#include <iostream>
using namespace std;

const int N = 1e6 + 10;
int n;
int q[N];

void qs(int *q, int l, int r) {
    if (l >= r) return ;
    int x = q[l], i = l - 1, j = r + 1;
    while (i < j) {
        while (q[++i] < x);
        while (q[--j] < x);
        if (i < j) swap(q[i], q[j]);
    }
    qs(q, l, j);
    qs(q, j + 1, r);
}

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &q[i]);
    qs(q, 0, n - 1);
    
    for (int i = 0; i < n; i++) printf("%d ", q[i]);
}
```

为啥循环条件不能等于呀? `do i++; while(a[i] <= x);`

> 会陷入死循环，比如2 1，2 <= 2、1 <= 2，0 <= 2，会不断的加。

为什么不能`qs(q, l, j - 1); qs(q, j, r);`?

> 当区间长度是2的时候，可能有边界问题。例如 2 1，那么会一直递归，因为区间一直没有压缩，也就是一直在[0,1]。可以写成qs(q, l, i - 1); qs(q,i, r);，但是相应的x=q[r]。

为什么不能`i=l, j=r`呢？

> 会出现边界问题，例如 99 51 84 48 76 42 6 6 38 10。会一直递归下去。

### 归并排序

```cpp
#include <iostream>
using namespace std;

const int N = 1e6 + 10;
int n;
int q[N], tmp[N];

void ms(int *q, int l, int r) {
    if (l >= r) return ;
    
    int mid = l + r >> 1;
    ms(q, l, mid), ms(q, mid + 1, r);
    int k = 0, i = l, j = mid + 1;
    
    while (i <= mid && j <= r) {
        if (q[i] <= q[j]) tmp[k++] = q[i++];
        else tmp[k++] = q[j++];
    }
    while (i <= mid) tmp[k++] = q[i++];
    while (j <= r) tmp[k++] = q[j++];
    
    for (i = l, j = 0; i <= r; i++, j++) q[i] = tmp[j];
}

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &q[i]);
    ms(q, 0, n - 1);
    
    for (int i = 0; i < n; i++) printf("%d ", q[i]);
}
```

```cpp
bool cmp(vector<int>& a, vector<int>& b) {
    if (a.size() != b.size()) return a.size() > b.size();
    for (int i = a.size() - 1; ~i; i--)  {
        if (a[i] != b[i]) return a[i] > b[i];

vector<int> sub(vector<int>& a, vector<int>& b) {
    vector<int> c;
    for (int i = 0, t = 0; i < a.size(); i++) {
        t = a[i] - t;
        if (i < b.size()) t -= b[i];
        c.emplace_back((t + 10) % 10);

        else t = 0;

    return c;
```

## 区间合并

```cpp
void merge(vector<PII> &segs)
{
    vector<PII> res;

    sort(segs.begin(), segs.end());

    int st = -2e9, ed = -2e9;
    for (auto seg : segs)
        if (ed < seg.first)
        {
            if (st != -2e9) res.push_back({st, ed});
            st = seg.first, ed = seg.second;
        }
        else ed = max(ed, seg.second);

    if (st != -2e9) res.push_back({st, ed});

    segs = res;
}
```

```cpp
void down(int u) {
    int t = u;
    if (u * 2 <= cnt && h[u * 2] < h[t]) t = u * 2;
    if (u * 2 + 1 <= cnt && h[u * 2 + 1] < h[t]) t = u * 2 + 1;
    if (t != u) {
        swap(h[u], h[t]);
        down(t);
    }

        u /= 2;
    }
```

```cpp
#include <iostream>
#include <cstring>
using namespace std;

typedef pair<int, int> PII;
const int N = 110;
int n, m;
int g[N][N];
int d[N][N];
PII q[N * N];

int bfs() {
    memset(d, -1, sizeof d);
    int hh = 0, tt = -1;
    q[++tt] = {0, 0};
    d[0][0] = 0;
    int dd[5] = {0, -1, 0, 1, 0};
    
    while (hh <= tt) {
        auto pre = q[hh++];
        for (int i = 0; i < 5; i++) {
            int x = pre.first + dd[i], y = pre.second + dd[i + 1];
            if (x >= 0 && x < n && y >= 0 && y < m && g[x][y] == 0 && d[x][y] == -1) {
                d[x][y] = d[pre.first][pre.second] + 1;
                q[++tt] = {x, y};
            }
        }
    }
    return d[n - 1][m - 1];
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> g[i][j];
        }
```

```cpp
void add(int a, int b) {
    e[idx] = b, ne[idx] = h[a], h[a] = idx++;
}

int topsort() {
    int hh = 0, tt = -1;
    for (int i = 1; i <= n; i++) {
        if (!d[i]) q[++tt] = i;
    }
    while (hh <= tt) {
        int t = q[hh++];
        for (int i = h[t]; ~i; i = ne[i]) {
            int j = e[i];
            d[j]--;
            if (!d[j]) q[++tt] = j;
        }
    }
    return tt == n - 1;
}

int main() {
    memset(h, -1, sizeof h);
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int a, b;
        cin >> a >> b;
        add(a, b);
        d[b]++;
    }
    
    if (topsort()) {
        for (int i = 0; i < n; i++) {
            printf("%d ", q[i]);
        }
    } else puts("-1");
}
```

```cpp
typedef pair<int, int> PII;

    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;
    priority_queue<PII, vector<PII>, greater<PII>> q;
    q.push({0, 1});

        auto p = q.top();
        q.pop();

                dist[j] = dis + w[i];
                q.push({dist[j], j});
            }
        }
    }

}
```

```cpp
int prim() {
    memset(dist, 0x3f, sizeof dist);

    for (int i = 0; i < n; i++) {
        int t = -1;
        for (int j = 1; j <= n; j++) {
```

```cpp
#include <iostream>
#include <cstring>
```

```cpp
const int N = 1e5 + 10, M = N * 2;

    st[u] = color;

        int j = e[i];
        if (!st[j]) {

}

                flag = false;

            }
        }
```
